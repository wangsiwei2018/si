WANG *chaRu(WANG *phead,int n)
{
	int i;
	WANG *pNow = phead->next;
	WANG *pLast = phead;
	WANG *p;
	for (i = 1; i < n; i++)
	{
		pLast = pNow;
		pNow = pNow->next;
	}
	p = (WANG*)malloc(sizeof(WANG));
	p->next = pNow->next;
	pLast->next->next=p;
	return p;
}

int number(char a[])
{
	int lena,i;
	lena=strlen(a);
	for(i=0;i<lena;i++)
		if(a[i]<'0'||a[i]>'9')
			return 0;
		return 1;
}
